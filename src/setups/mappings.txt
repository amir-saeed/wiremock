"""Deploy WireMock image to ECR and trigger ECS redeployment."""

import base64
import logging
import sys
from dataclasses import dataclass

import boto3
import docker
from botocore.exceptions import ClientError, NoCredentialsError

logging.basicConfig(level=logging.INFO, format="%(asctime)s [%(levelname)s] %(message)s")
logger = logging.getLogger(__name__)


@dataclass(frozen=True)
class Config:
    aws_region: str = "us-east-1"
    ecr_repo_uri: str = "123456789012.dkr.ecr.us-east-1.amazonaws.com/my-wiremock"
    ecs_cluster: str = "wiremock-cluster"
    ecs_service: str = "wiremock-service"
    podman_uri: str = "tcp://127.0.0.1:2375"
    image_tag: str = "latest"


def get_podman_client(uri: str) -> docker.DockerClient:
    client = docker.DockerClient(base_url=uri)
    client.ping()
    logger.info("Connected to Podman at %s", uri)
    return client


def ecr_login(client: docker.DockerClient, region: str, repo_uri: str) -> None:
    ecr = boto3.client("ecr", region_name=region)
    token = ecr.get_authorization_token()["authorizationData"][0]["authorizationToken"]
    username, password = base64.b64decode(token).decode().split(":")
    registry = repo_uri.split("/")[0]
    client.login(username=username, password=password, registry=registry)
    logger.info("Authenticated with ECR")


def build_and_push(client: docker.DockerClient, repo_uri: str, tag: str) -> None:
    full_tag = f"{repo_uri}:{tag}"
    logger.info("Building image: %s", full_tag)
    client.images.build(path=".", tag=full_tag, rm=True)

    logger.info("Pushing to ECR...")
    for log in client.images.push(repo_uri, tag=tag, stream=True, decode=True):
        if "status" in log:
            logger.info(log["status"])
        if "error" in log:
            raise RuntimeError(log["error"])


def redeploy_ecs(region: str, cluster: str, service: str) -> None:
    ecs = boto3.client("ecs", region_name=region)
    ecs.update_service(cluster=cluster, service=service, forceNewDeployment=True)
    logger.info("ECS redeployment triggered: %s/%s", cluster, service)


def main() -> None:
    cfg = Config()
    try:
        client = get_podman_client(cfg.podman_uri)
        ecr_login(client, cfg.aws_region, cfg.ecr_repo_uri)
        build_and_push(client, cfg.ecr_repo_uri, cfg.image_tag)
        redeploy_ecs(cfg.aws_region, cfg.ecs_cluster, cfg.ecs_service)
    except NoCredentialsError:
        logger.error("AWS credentials not found. Run 'aws-azure-login' first.")
        sys.exit(1)
    except ClientError as e:
        logger.error("AWS error: %s", e.response["Error"]["Message"])
        sys.exit(1)
    except docker.errors.APIError as e:
        logger.error("Podman error: %s", e.explanation)
        sys.exit(1)


if __name__ == "__main__":
    main()
