def test_main_metastore_defaults_to_none_when_absent(monkeypatch) -> None:
    args = {k: v for k, v in _base_args().items() if k != "METASTORE"}
    mocks = _setup_main(monkeypatch)
    monkeypatch.setattr(matching_job, "parse_args", MagicMock(return_value=args))
    matching_job.main()

    runtime_args = mocks.load_job_context.call_args.kwargs["glue_runtime_args"]
    assert runtime_args["metastore"] is None
