def parse_execution_ts(value: str) -> datetime:
    """Parses execution_ts string in YYYYMMDDHHmmss format, returns UTC datetime.

    Example valid value: '20260615143022'
    """
    try:
        dt = datetime.strptime(value, "%Y%m%d%H%M%S")
    except ValueError:
        raise ValueError(
            f"Invalid execution_ts '{value}': expected YYYYMMDDHHmmss format (e.g. '20260615143022')"
        )
    return dt.replace(tzinfo=UTC)
