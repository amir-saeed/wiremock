"""Tests for src/jobs/matching_job.py entrypoint.

matching_job imports getResolvedOptions inside parse_args() (not at module level),
so we cannot use the maintenance_job pattern of monkeypatching matching_job.getResolvedOptions.
Instead we patch sys.modules["awsglue.utils"].getResolvedOptions directly — the conftest
already stubs that module, so the attribute is always patchable.

Key changes from old version:
  - CONFIG_BASE_PATH removed from REQUIRED_ARGS (Step 2/3)
  - EXECUTION_TS and METASTORE added to REQUIRED_ARGS (Step 4)
  - OPTIONAL_ARGS is now only ES_PORT (not AWS_REGION / ES_SECRET_NAME)
  - aws_region comes from resolve_aws_region(), not a CLI arg
  - No config_base_path or secret_name in glue_runtime_args
"""
import sys
from datetime import UTC, date, datetime
from types import SimpleNamespace
from unittest.mock import MagicMock

import jobs.matching_job as matching_job
from test import constants

METASTORE = "arn:aws:glue:eu-west-1:123456789012:catalog"


# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------

def _base_args(overrides: dict | None = None) -> dict:
    """Minimal valid args dict matching REQUIRED_ARGS."""
    args = {
        "JOB_SUBTYPE": "person",
        "ENV": "local",
        "REPORTING_DATE": "2026-05-12",
        "EXECUTION_TS": constants.EXECUTION_TS,
        "ES_HOST": "localhost",
        "ES_PORT": "5432",
        "ES_DATABASE": "entity_store",
        "ES_USER": "motor_eqd",
        "METASTORE": METASTORE,
    }
    if overrides:
        args.update(overrides)
    return args


def _make_get_resolved_options(provided_args: dict):
    """Returns a fake getResolvedOptions that silently omits absent keys."""
    def _fake(_argv, keys):
        return {k: provided_args[k] for k in keys if k in provided_args}
    return _fake


def _setup_main(monkeypatch, args_overrides: dict | None = None):
    """Patches parse_args, resolve_aws_region, PIPELINES, and load_job_context.

    Returns a SimpleNamespace(pipeline, pipeline_cls, pipeline_spec, load_job_context).
    """
    pipeline = MagicMock()
    pipeline_cls = MagicMock(return_value=pipeline)
    pipeline_spec = SimpleNamespace(
        pipeline_cls=pipeline_cls,
        job_context_model=object,
    )
    load_job_context = MagicMock(return_value=MagicMock())

    monkeypatch.setattr(
        matching_job,
        "parse_args",
        MagicMock(return_value=_base_args(args_overrides)),
    )
    monkeypatch.setattr(
        matching_job,
        "resolve_aws_region",
        MagicMock(return_value=constants.AWS_REGION),
    )
    monkeypatch.setattr(
        matching_job,
        "PIPELINES",
        {(matching_job.JobName.MATCHING, matching_job.VAP.PERSON): pipeline_spec},
    )
    monkeypatch.setattr(matching_job, "load_job_context", load_job_context)

    return SimpleNamespace(
        pipeline=pipeline,
        pipeline_cls=pipeline_cls,
        pipeline_spec=pipeline_spec,
        load_job_context=load_job_context,
    )


# ---------------------------------------------------------------------------
# parse_args
# ---------------------------------------------------------------------------

def test_parse_args_returns_all_required_keys(monkeypatch) -> None:
    monkeypatch.setattr(
        sys.modules["awsglue.utils"],
        "getResolvedOptions",
        _make_get_resolved_options(_base_args()),
    )

    result = matching_job.parse_args()

    for key in matching_job.REQUIRED_ARGS:
        assert key in result


def test_parse_args_includes_optional_es_port_when_provided(monkeypatch) -> None:
    monkeypatch.setattr(
        sys.modules["awsglue.utils"],
        "getResolvedOptions",
        _make_get_resolved_options({**_base_args(), "ES_PORT": "5433"}),
    )

    result = matching_job.parse_args()

    assert result["ES_PORT"] == "5433"


def test_parse_args_silently_skips_optional_arg_when_it_raises(monkeypatch) -> None:
    call_count = {"n": 0}

    def _raise_on_optional(_argv, keys):
        call_count["n"] += 1
        # First call is for REQUIRED_ARGS (bulk); subsequent single-key calls are optional
        if call_count["n"] > 1 and keys == ["ES_PORT"]:
            raise RuntimeError("optional arg not provided")
        return {k: _base_args().get(k) for k in keys if k in _base_args()}

    monkeypatch.setattr(
        sys.modules["awsglue.utils"],
        "getResolvedOptions",
        _raise_on_optional,
    )

    # Should not propagate the RuntimeError
    result = matching_job.parse_args()
    assert "JOB_SUBTYPE" in result
