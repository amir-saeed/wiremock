"""Tests for src/utils/s3.py — check_dap_s3_not_exist."""
from datetime import date
from unittest.mock import MagicMock

import pytest

from utils.exceptions import AlreadyExistsError
from utils.s3 import DAP_S3_PREFIX_TEMPLATE, check_dap_s3_not_exist

REPORTING_DATE = date(2026, 5, 12)
BUCKET = "b4c-dal-local-data-quote-enrichment"
PREFIX = DAP_S3_PREFIX_TEMPLATE.format(date=REPORTING_DATE)


def _make_s3(key_count: int = 0) -> MagicMock:
    s3 = MagicMock()
    s3.list_objects_v2.return_value = {"KeyCount": key_count}
    return s3


# ---------------------------------------------------------------------------
# Happy path
# ---------------------------------------------------------------------------

def test_passes_when_no_objects_exist() -> None:
    check_dap_s3_not_exist(
        reporting_date=REPORTING_DATE,
        s3=_make_s3(key_count=0),
        bucket=BUCKET,
        logger=MagicMock(),
    )


def test_calls_list_objects_v2_with_correct_args() -> None:
    s3 = _make_s3()

    check_dap_s3_not_exist(
        reporting_date=REPORTING_DATE,
        s3=s3,
        bucket=BUCKET,
        logger=MagicMock(),
    )

    s3.list_objects_v2.assert_called_once_with(
        Bucket=BUCKET,
        Prefix=PREFIX,
        MaxKeys=1,
    )


def test_uses_correct_prefix_format() -> None:
    s3 = _make_s3()

    check_dap_s3_not_exist(
        reporting_date=REPORTING_DATE,
        s3=s3,
        bucket=BUCKET,
        logger=MagicMock(),
    )

    _, kwargs = s3.list_objects_v2.call_args
    assert kwargs["Prefix"] == "reporting_date=2026-05-12/"


# ---------------------------------------------------------------------------
# Duplicate detected
# ---------------------------------------------------------------------------

def test_raises_when_objects_exist() -> None:
    with pytest.raises(AlreadyExistsError):
        check_dap_s3_not_exist(
            reporting_date=REPORTING_DATE,
            s3=_make_s3(key_count=1),
            bucket=BUCKET,
            logger=MagicMock(),
        )


def test_error_reporting_date_matches() -> None:
    with pytest.raises(AlreadyExistsError) as exc_info:
        check_dap_s3_not_exist(
            reporting_date=REPORTING_DATE,
            s3=_make_s3(key_count=1),
            bucket=BUCKET,
            logger=MagicMock(),
        )

    assert exc_info.value.reporting_date == REPORTING_DATE


def test_error_location_contains_bucket_and_prefix() -> None:
    with pytest.raises(AlreadyExistsError) as exc_info:
        check_dap_s3_not_exist(
            reporting_date=REPORTING_DATE,
            s3=_make_s3(key_count=1),
            bucket=BUCKET,
            logger=MagicMock(),
        )

    assert exc_info.value.location == f"s3://{BUCKET}/{PREFIX}"


def test_key_count_zero_does_not_raise() -> None:
    s3 = MagicMock()
    s3.list_objects_v2.return_value = {"KeyCount": 0}

    check_dap_s3_not_exist(
        reporting_date=REPORTING_DATE,
        s3=s3,
        bucket=BUCKET,
        logger=MagicMock(),
    )


def test_missing_key_count_in_response_does_not_raise() -> None:
    """Response without KeyCount key defaults to 0 — no error."""
    s3 = MagicMock()
    s3.list_objects_v2.return_value = {}

    check_dap_s3_not_exist(
        reporting_date=REPORTING_DATE,
        s3=s3,
        bucket=BUCKET,
        logger=MagicMock(),
    )
