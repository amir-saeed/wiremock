"""Tests for src/pipelines/maintenance/maintenance_pipeline.py.

Strategy
--------
`execute` is the only non-trivial code path. We call it directly on a
MaintenancePipeline instance — bypassing `run()` entirely so we never touch
GlueContext or SparkContext (both need a real Spark cluster).

The four step modules (retention, partitions, duplicate_check, ingestion)
are patched at the *pipeline* module's namespace so the patch applies
regardless of how the module was imported.
"""
from __future__ import annotations

from datetime import UTC, date, datetime
from types import SimpleNamespace
from unittest.mock import MagicMock, call, patch

import pytest

from pipelines.maintenance.maintenance_pipeline import MaintenancePipeline
from test import constants
from utils.enums import Env, JobName

# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------

REPORTING_DATE = date(2026, 5, 12)
EXECUTION_TS = datetime(2026, 5, 12, 9, 0, 0, tzinfo=UTC)

PIPELINE_MODULE = "pipelines.maintenance.maintenance_pipeline"


def _make_glue_runtime() -> SimpleNamespace:
    return SimpleNamespace(
        job_name=JobName.MAINTENANCE,
        env=Env.LOCAL,
        reporting_date=REPORTING_DATE,
        execution_ts=EXECUTION_TS,
        metastore=constants.METASTORE,
        config_s3=constants.CONFIG_S3,
        dap_s3=constants.DAP_S3,
    )


def _make_job_context() -> MagicMock:
    ctx = MagicMock()
    ctx.glue_runtime = _make_glue_runtime()
    return ctx


def _make_resources() -> MagicMock:
    resources = MagicMock()
    resources.postgres = MagicMock()
    resources.spark = MagicMock()
    return resources


def _make_pipeline() -> MaintenancePipeline:
    pipeline = MaintenancePipeline()
    pipeline.logger = MagicMock()
    return pipeline


# ---------------------------------------------------------------------------
# _build_glue_args
# ---------------------------------------------------------------------------

def test_build_glue_args_contains_job_name() -> None:
    ctx = _make_job_context()
    result = MaintenancePipeline._build_glue_args(ctx)
    assert result["JOB_NAME"] == JobName.MAINTENANCE.value


def test_build_glue_args_contains_reporting_date() -> None:
    ctx = _make_job_context()
    result = MaintenancePipeline._build_glue_args(ctx)
    assert result["REPORTING_DATE"] == str(REPORTING_DATE)


def test_build_glue_args_contains_env() -> None:
    ctx = _make_job_context()
    result = MaintenancePipeline._build_glue_args(ctx)
    assert result["ENV"] == Env.LOCAL.value


def test_build_glue_args_contains_execution_ts() -> None:
    ctx = _make_job_context()
    result = MaintenancePipeline._build_glue_args(ctx)
    assert result["EXECUTION_TS"] == EXECUTION_TS.isoformat()


def test_build_glue_args_returns_exactly_four_keys() -> None:
    ctx = _make_job_context()
    result = MaintenancePipeline._build_glue_args(ctx)
    assert set(result.keys()) == {"JOB_NAME", "REPORTING_DATE", "ENV", "EXECUTION_TS"}

# execute — step ordering (retention → partitions → duplicate_check → ingestion)
# ---------------------------------------------------------------------------

@patch(f"{PIPELINE_MODULE}.retention")
@patch(f"{PIPELINE_MODULE}.partitions")
@patch(f"{PIPELINE_MODULE}.duplicate_check")
@patch(f"{PIPELINE_MODULE}.ingestion")
def test_execute_steps_run_in_order(ingestion, duplicate_check, partitions, retention) -> None:
    """Steps must run in the documented order: retention → partitions → dupe check → ingestion."""
    manager = MagicMock()
    manager.attach_mock(retention.apply_retention, "retention")
    manager.attach_mock(partitions.ensure_partitions, "partitions")
    manager.attach_mock(duplicate_check.check_quotes_not_exist, "duplicate_check")
    manager.attach_mock(ingestion.ingest_quotes, "ingestion")

    pipeline = _make_pipeline()
    pipeline.execute(_make_job_context(), _make_resources())

    call_names = [c[0] for c in manager.mock_calls]
    assert call_names == ["retention", "partitions", "duplicate_check", "ingestion"]
