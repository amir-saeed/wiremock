"""Tests for jobs/maintenance_job.py entrypoint.

maintenance_job is imported at module level (not inside each test function) so
that the awsglue stub in conftest.py is already in sys.modules when the import
runs. This guarantees maintenance_job.getResolvedOptions exists as a MagicMock
and monkeypatch.setattr can target it reliably.

All 11 args are required — no OPTIONAL_ARGS, no default fallbacks.
JOB_SUBTYPE is parsed but not forwarded to glue_runtime_args (maintenance
has no subtype; PIPELINES key is always (MAINTENANCE, None)).
"""
from datetime import UTC, date, datetime
from types import SimpleNamespace
from unittest.mock import MagicMock

import jobs.maintenance_job as maintenance_job
from test import constants

# ---------------------------------------------------------------------------
# Expected args — mirrors the list in parse_args()
# ---------------------------------------------------------------------------

EXPECTED_ARG_KEYS = [
    "JOB_SUBTYPE",
    "ENV",
    "REPORTING_DATE",
    "EXECUTION_TS",
    "METASTORE",
    "CONFIG_S3",
    "DAP_S3",
    "ES_HOST",
    "ES_PORT",
    "ES_DATABASE",
    "ES_USER",
]


# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------

def _base_args(overrides: dict | None = None) -> dict:
    """Minimal valid args dict matching all required args."""
    args = {
        "JOB_SUBTYPE": "person",
        "ENV": "local",
        "REPORTING_DATE": "2026-05-12",
        "EXECUTION_TS": constants.EXECUTION_TS,
        "METASTORE": constants.METASTORE,
        "CONFIG_S3": constants.CONFIG_S3,
        "DAP_S3": constants.DAP_S3,
        "ES_HOST": "localhost",
        "ES_PORT": "5432",
        "ES_DATABASE": "entity_store",
        "ES_USER": "motor_eqd",
    }
    if overrides:
        args.update(overrides)
    return args


def _setup_main(monkeypatch, args_overrides: dict | None = None):
    """Patches configure_logging, parse_args, PIPELINES, and load_job_context.

    Returns a SimpleNamespace(pipeline, pipeline_cls, pipeline_spec, load_job_context).
    """
    pipeline = MagicMock()
    pipeline_cls = MagicMock(return_value=pipeline)
    pipeline_spec = SimpleNamespace(
        pipeline_cls=pipeline_cls,
        job_context_model=object,
    )
    load_job_context = MagicMock(return_value=MagicMock())

    monkeypatch.setattr(
        maintenance_job,
        "configure_logging",
        MagicMock(return_value=MagicMock()),
    )
    monkeypatch.setattr(
        maintenance_job,
        "parse_args",
        MagicMock(return_value=_base_args(args_overrides)),
    )
    monkeypatch.setattr(
        maintenance_job,
        "PIPELINES",
        {(maintenance_job.JobName.MAINTENANCE, None): pipeline_spec},
    )
    monkeypatch.setattr(maintenance_job, "load_job_context", load_job_context)

    return SimpleNamespace(
        pipeline=pipeline,
        pipeline_cls=pipeline_cls,
        pipeline_spec=pipeline_spec,
        load_job_context=load_job_context,
    )


# ---------------------------------------------------------------------------
# parse_args
# ---------------------------------------------------------------------------

def test_parse_args_returns_all_required_keys(monkeypatch) -> None:
    provided = _base_args()
    monkeypatch.setattr(
        maintenance_job.getResolvedOptions,
        "side_effect",
        lambda argv, keys: {k: provided[k] for k in keys},
    )

    result = maintenance_job.parse_args()

    for key in EXPECTED_ARG_KEYS:
        assert key in result


# ---------------------------------------------------------------------------
# main() — runtime args construction
# ---------------------------------------------------------------------------

def test_main_sets_job_name_to_maintenance(monkeypatch) -> None:
    mocks = _setup_main(monkeypatch)
    maintenance_job.main()

    runtime_args = mocks.load_job_context.call_args.kwargs["glue_runtime_args"]
    assert runtime_args["job_name"] == "maintenance"
