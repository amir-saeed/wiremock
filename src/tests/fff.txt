EXECUTION_TS_FORMAT = "%Y%m%d%H%M%S"

def parse_execution_ts(value: str, exec_ts_format: str = EXECUTION_TS_FORMAT) -> datetime:
    """Parses execution_ts string, returns UTC datetime.

    Example valid value: '20260615143022'
    """
    dt = parse_date(value, exec_ts_format, "datetime")
    return dt.replace(tzinfo=UTC)  # type: ignore[union-attr]
