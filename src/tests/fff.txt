from datetime import UTC, date, datetime

import pytest
from pydantic import ValidationError
from test.constants import EXECUTION_TS, REPORTING_DATE, S3_PATH

from utils.enums import Env, JobName
from utils.models import (
    MaintenanceJobConfig,
    MaintenanceJobContext,
    MaintenanceJobRuntime,
)


# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------

def _valid_es() -> dict:
    return {
        "host": "localhost",
        "port": 5432,
        "database": "entity_store",
        "user": "motor_eqd",
        "secret_name": None,
    }


def _valid_source_datasets() -> dict:
    return {
        "quote": {
            "path": f"{S3_PATH}/quote",
            "columns": ["correlationid", "channel", "source_timestamp"],
        }
    }


def _make_maintenance_runtime(overrides: dict | None = None) -> MaintenanceJobRuntime:
    fields = {
        "job_name": JobName.MAINTENANCE,
        "env": Env.LOCAL,
        "aws_region": "eu-west-1",
        "config_base_path": "s3://bucket/config",
        "reporting_date": REPORTING_DATE,
        "execution_ts": EXECUTION_TS,
        "es": _valid_es(),
    }
    if overrides:
        fields.update(overrides)
    return MaintenanceJobRuntime(**fields)


def _make_maintenance_config(overrides: dict | None = None) -> MaintenanceJobConfig:
    fields = {
        "dap_enrichment_bucket": "b4c-dal-local-data-quote-enrichment",
        "source_datasets": _valid_source_datasets(),
    }
    if overrides:
        fields.update(overrides)
    return MaintenanceJobConfig(**fields)


# ---------------------------------------------------------------------------
# MaintenanceJobRuntime
# ---------------------------------------------------------------------------

def test_maintenance_job_runtime_parses_date_and_datetime_fields() -> None:
    runtime = _make_maintenance_runtime()

    assert runtime.reporting_date == date(2026, 5, 12)
    assert runtime.execution_ts == datetime(2026, 5, 12, 9, tzinfo=UTC)
    assert runtime.job_name == JobName.MAINTENANCE
    assert runtime.env == Env.LOCAL


def test_maintenance_job_runtime_empty_aws_region_raises() -> None:
    with pytest.raises(ValidationError):
        _make_maintenance_runtime({"aws_region": ""})


def test_maintenance_job_runtime_whitespace_aws_region_raises() -> None:
    with pytest.raises(ValidationError):
        _make_maintenance_runtime({"aws_region": "   "})


def test_maintenance_job_runtime_empty_config_base_path_raises() -> None:
    with pytest.raises(ValidationError):
        _make_maintenance_runtime({"config_base_path": ""})


# ---------------------------------------------------------------------------
# MaintenanceJobConfig — data_retention_days
# ---------------------------------------------------------------------------

def test_maintenance_job_config_default_retention_days_is_1095() -> None:
    config = _make_maintenance_config()
    assert config.data_retention_days == 1095


@pytest.mark.parametrize("days", [1000, 1001, 1095, 1461, 3650])
def test_maintenance_job_config_valid_retention_days(days: int) -> None:
    config = _make_maintenance_config({"data_retention_days": days})
    assert config.data_retention_days == days


@pytest.mark.parametrize("days", [999, 0, -1, -9999, 1])
def test_maintenance_job_config_invalid_retention_days_raises(days: int) -> None:
    with pytest.raises(ValidationError):
        _make_maintenance_config({"data_retention_days": days})


def test_maintenance_job_config_retention_days_error_mentions_field() -> None:
    with pytest.raises(ValidationError) as exc_info:
        _make_maintenance_config({"data_retention_days": 500})
    assert "data_retention_days" in str(exc_info.value)

def test_maintenance_job_context_resolves_env_placeholders() -> None:
    context = MaintenanceJobContext(
        glue_runtime={
            "job_name": JobName.MAINTENANCE.value,
            "env": Env.LOCAL.value,
            "aws_region": "eu-west-1",
            "config_base_path": "s3://bucket/glue/config",
            "reporting_date": REPORTING_DATE,
            "execution_ts": EXECUTION_TS,
            "es": _valid_es(),
        },
        job={
            "dap_enrichment_bucket": "b4c-dal-{env}-data-quote-enrichment",
            "source_datasets": {
                "quote": {
                    "path": f"{S3_PATH}/{{env}}/quote",
                    "columns": ["correlationid", "channel", "source_timestamp"],
                }
            },
        },
    )

    assert context.job.source_datasets["quote"].path == f"{S3_PATH}/local/quote"
    assert context.job.dap_enrichment_bucket == "b4c-dal-local-data-quote-enrichment"
    assert context.env == Env.LOCAL
