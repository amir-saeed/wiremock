# Module-level helper: job field annotated as str, not BaseJobConfig — used in TypeError test.
class _WrongJobContext(BaseJobContext):
    job: str


# ---------------------------------------------------------------------------
# _read_local_yaml
# ---------------------------------------------------------------------------

def test_read_local_yaml_returns_dict_for_valid_yaml_file(tmp_path: Path) -> None:
    config_file = tmp_path / "config.yaml"
    config_file.write_text(yaml.safe_dump(MATCHING_JOB_CONFIG))

    result = _read_local_yaml(config_file)

    assert result == MATCHING_JOB_CONFIG


def test_read_local_yaml_returns_empty_dict_for_empty_file(tmp_path: Path) -> None:
    # yaml.safe_load of an empty file returns None; the `or {}` guard must convert it
    empty_file = tmp_path / "empty.yaml"
    empty_file.write_text("")

    result = _read_local_yaml(empty_file)

    assert result == {}


def test_read_local_yaml_raises_file_not_found_for_missing_file(tmp_path: Path) -> None:
    missing = tmp_path / "nonexistent.yaml"

    with pytest.raises(FileNotFoundError, match="nonexistent.yaml"):
        _read_local_yaml(missing)


def test_read_local_yaml_error_message_includes_full_path(tmp_path: Path) -> None:
    missing = tmp_path / "jobs" / "matching" / "person.yaml"

    with pytest.raises(FileNotFoundError, match="Job config file not found"):
        _read_local_yaml(missing)


def test_read_local_yaml_returns_nested_dict(tmp_path: Path) -> None:
    nested = {"level1": {"level2": {"key": "value"}}}
    config_file = tmp_path / "nested.yaml"
    config_file.write_text(yaml.safe_dump(nested))

    assert _read_local_yaml(config_file) == nested


# ---------------------------------------------------------------------------
# _job_config_path
# ---------------------------------------------------------------------------

def test_job_config_path_with_subtype_uses_subtype_as_filename() -> None:
    path = _job_config_path("matching", "person")

    assert path.parts[-1] == "person.yaml"
    assert path.parts[-2] == "matching"
    assert path.parts[-3] == "jobs"


def test_job_config_path_without_subtype_uses_job_name_as_filename() -> None:
    path = _job_config_path("maintenance", None)

    assert path.parts[-1] == "maintenance.yaml"
    assert path.parts[-2] == "maintenance"
    assert path.parts[-3] == "jobs"


def test_job_config_path_returns_path_object() -> None:
    assert isinstance(_job_config_path("matching", "person"), Path)


def test_job_config_path_different_subtypes_produce_different_paths() -> None:
    person = _job_config_path("matching", "person")
    vehicle = _job_config_path("matching", "vehicle")

    assert person != vehicle
    assert person.parts[-1] == "person.yaml"
    assert vehicle.parts[-1] == "vehicle.yaml"

def test_resolve_field_model_returns_expected_subclass_for_job() -> None:

        _resolve_field_model(_WrongJobContext, "job", BaseJobConfig)
