def test_delete_quotes_returns_zero_when_first_batch_is_empty(mock_psycopg_connect) -> None:
    """rowcount=0 on first execute → break immediately, total=0."""
    mock_psycopg_connect.cursor.rowcount = 0
    result = _make_connector().delete_quotes_before_date(cutoff_date=date(2023, 5, 13))
    assert result == 0


def test_delete_quotes_accumulates_across_batches(mock_psycopg_connect) -> None:
    """rowcount=[10, 0] → one real batch then stop, total=10."""
    cursor = mock_psycopg_connect.cursor
    rowcounts = iter([10, 0])
    cursor.execute.side_effect = lambda *_: setattr(cursor, "rowcount", next(rowcounts))

    result = _make_connector().delete_quotes_before_date(cutoff_date=date(2023, 5, 13))
    assert result == 10
