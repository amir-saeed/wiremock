
#!/bin/bash
# Run maintenance job locally against glue-test2 container
# Usage: bash scripts/run_maintenance_local.sh 2026-06-26
 
REPORTING_DATE=${1:-2026-06-26}
EXECUTION_TS=$(date -d "$REPORTING_DATE" +"%Y%m%d100000" 2>/dev/null || echo "20260626100000")
 
podman exec \
  -e AWS_DEFAULT_REGION=eu-west-1 \
  -e ES_PASSWORD=motor_eqd \
  -e PYTHONPATH=/tmp/glue/src:/usr/share/aws/glue-pds \
  -e AWS_ACCESS_KEY_ID=test \
  -e AWS_SECRET_ACCESS_KEY=test \
  -e AWS_ENDPOINT_URL_DYNAMODB=http://10.72.95.111:8000 \
  -e AWS_ENDPOINT_URL_S3=http://10.72.95.111:9090 \
  -e AWS_S3_FORCE_PATH_STYLE=true \
  glue-test2 python /tmp/glue/jobs/maintenance_job.py \
  --JOB_SUBTYPE person \
  --ENV local \
  --REPORTING_DATE "$REPORTING_DATE" \
  --EXECUTION_TS "$EXECUTION_TS" \
  --METASTORE arn:aws:glue:eu-west-1:123456789012:catalog \
  --CONFIG_S3 s3://b4c-dal-local-data-quotes-shredded-lfs/config \
  --DAP_S3 s3://b4c-dal-local-data-quote-enrichment \
  --ES_HOST 10.72.95.111 \
  --ES_PORT 5432 \
  --ES_DATABASE b4c-local-entity-store \
  --ES_USER motor_eqd
