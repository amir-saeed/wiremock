def _base_args(overrides: dict | None = None) -> dict:
    """Minimal valid args dict matching REQUIRED_ARGS + AWS_REGION."""
    args = {
        "ENV": "local",
        "REPORTING_DATE": "2026-05-12",
        "EXECUTION_TS": constants.EXECUTION_TS,
        "CONFIG_BASE_PATH": "s3://bucket/config",
        "ES_HOST": "localhost",
        "ES_DATABASE": "entity_store",
        "ES_USER": "motor_eqd",
        "AWS_REGION": constants.AWS_REGION,
    }
    if overrides:
        args.update(overrides)
    return args


def _make_get_resolved_options(provided_args: dict):
    """Returns a fake getResolvedOptions that serves from provided_args,
    raising for any key not present (simulates missing optional args)."""

    def _fake(argv, keys):
        missing = [k for k in keys if k not in provided_args]
        if missing:
            raise SystemExit(f"No option found for {missing[0]}")
        return {k: provided_args[k] for k in keys}

    return _fake


def _setup_main(monkeypatch, args_overrides: dict | None = None):
    """Patches parse_args, PIPELINES, and load_job_context for main() tests.

    Returns a SimpleNamespace(pipeline, pipeline_cls, pipeline_spec, load_job_context).
    """
    import jobs.maintenance_job as maintenance_job

    pipeline = MagicMock()
    pipeline_cls = MagicMock(return_value=pipeline)
    pipeline_spec = SimpleNamespace(
        pipeline_cls=pipeline_cls,
        job_context_model=object,
    )
    load_job_context = MagicMock(return_value=MagicMock())

    monkeypatch.setattr(
        maintenance_job,
        "parse_args",
        MagicMock(return_value=_base_args(args_overrides)),
    )
    monkeypatch.setattr(
        maintenance_job,
        "PIPELINES",
        {(maintenance_job.JobName.MAINTENANCE, None): pipeline_spec},
    )
    monkeypatch.setattr(maintenance_job, "load_job_context", load_job_context)

    return SimpleNamespace(
        pipeline=pipeline,
        pipeline_cls=pipeline_cls,
        pipeline_spec=pipeline_spec,
        load_job_context=load_job_context,
    )


# ---------------------------------------------------------------------------
# parse_args tests
# ---------------------------------------------------------------------------

def test_parse_args_returns_all_required_keys(monkeypatch) -> None:
    import jobs.maintenance_job as maintenance_job

    monkeypatch.setattr(
        maintenance_job,
        "getResolvedOptions",
        _make_get_resolved_options(_base_args()),
    )

    result = maintenance_job.parse_args()

    for key in maintenance_job.REQUIRED_ARGS:
        assert key in result


def test_parse_args_includes_optional_args_when_provided(monkeypatch) -> None:
    import jobs.maintenance_job as maintenance_job

    full_args = {
        **_base_args(),
        "ES_PORT": "5433",
        "ES_SECRET_NAME": "my-secret",
        "DATA_RETENTION_DAYS": "730",
    }
    monkeypatch.setattr(
        maintenance_job,
        "getResolvedOptions",
        _make_get_resolved_options(full_args),
    )

    result = maintenance_job.parse_args()

    assert result["ES_PORT"] == "5433"
    assert result["ES_SECRET_NAME"] == "my-secret"
    assert result["DATA_RETENTION_DAYS"] == "730"


def test_parse_args_silently_skips_absent_optional_args(monkeypatch) -> None:
    import jobs.maintenance_job as maintenance_job

    # Only required args are available — optionals will raise inside getResolvedOptions
    monkeypatch.setattr(
        maintenance_job,
        "getResolvedOptions",
        _make_get_resolved_options(_base_args()),
    )

    result = maintenance_job.parse_args()

    for key in maintenance_job.OPTIONAL_ARGS:
        assert key not in result


def test_parse_args_some_optional_args_present(monkeypatch) -> None:
    """Only AWS_REGION provided among optionals — others must be silently absent."""
    import jobs.maintenance_job as maintenance_job

    partial_args = {**_base_args(), "AWS_REGION": constants.AWS_REGION}
    monkeypatch.setattr(
        maintenance_job,
        "getResolvedOptions",
        _make_get_resolved_options(partial_args),
    )

    result = maintenance_job.parse_args()

    assert result["AWS_REGION"] == constants.AWS_REGION
    assert "ES_PORT" not in result
    assert "DATA_RETENTION_DAYS" not in result
