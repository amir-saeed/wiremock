"""Tests for src/pipelines/maintenance/duplicate_check.py."""
from datetime import date
from unittest.mock import MagicMock, call

import pytest

from pipelines.maintenance import duplicate_check
from tests import constants

# ---------------------------------------------------------------------------
# Shared fixtures
# ---------------------------------------------------------------------------

REPORTING_DATE = date(2026, 5, 12)
ENV = "local"
DAP_BUCKET = "b4c-dal-local-data-quote-enrichment"


def _make_dynamodb(count: int = 0) -> MagicMock:
    """Returns a mocked DynamoDB client where every query returns Count=count."""
    dynamodb = MagicMock()
    dynamodb.query.return_value = {"Count": count}
    return dynamodb


# ---------------------------------------------------------------------------
# check_quotes_not_exist
# ---------------------------------------------------------------------------

def test_check_quotes_not_exist_passes_when_no_quotes() -> None:
    postgres = MagicMock()
    postgres.has_quotes_for_reporting_date.return_value = False

    duplicate_check.check_quotes_not_exist(
        reporting_date=REPORTING_DATE,
        postgres=postgres,
        logger=MagicMock(),
    )

    postgres.has_quotes_for_reporting_date.assert_called_once_with(
        reporting_date=REPORTING_DATE,
    )


def test_check_quotes_not_exist_raises_when_quotes_exist() -> None:
    postgres = MagicMock()
    postgres.has_quotes_for_reporting_date.return_value = True

    with pytest.raises(AlreadyExistsError) as exc_info:
        duplicate_check.check_quotes_not_exist(
            reporting_date=REPORTING_DATE,
            postgres=postgres,
            logger=MagicMock(),
        )

    assert "2026-05-12" in str(exc_info.value)
    assert "Re-run requires explicit cleanup first" in str(exc_info.value)


def test_check_quotes_not_exist_error_mentions_quote_table() -> None:
    postgres = MagicMock()
    postgres.has_quotes_for_reporting_date.return_value = True

    with pytest.raises(AlreadyExistsError) as exc_info:

            reporting_date=REPORTING_DATE,

        )

def test_check_dap_s3_not_exist_passes_when_bucket_empty() -> None:
    s3 = MagicMock()
    s3.list_objects_v2.return_value = {"KeyCount": 0}

    duplicate_check.check_dap_s3_not_exist(
        reporting_date=REPORTING_DATE,
        s3=s3,
        bucket=DAP_BUCKET,
        logger=MagicMock(),
    )

    s3.list_objects_v2.assert_called_once_with(
        Bucket=DAP_BUCKET,
        Prefix="reporting_date=2026-05-12/",
        MaxKeys=1,
    )


def test_check_dap_s3_not_exist_raises_when_objects_exist() -> None:
    s3 = MagicMock()
    s3.list_objects_v2.return_value = {"KeyCount": 1}

    with pytest.raises(AlreadyExistsError) as exc_info:
        duplicate_check.check_dap_s3_not_exist(
            reporting_date=REPORTING_DATE,
            s3=s3,
            bucket=DAP_BUCKET,
            logger=MagicMock(),
        )

    assert "reporting_date=2026-05-12/" in str(exc_info.value)
    assert DAP_BUCKET in str(exc_info.value)


def test_check_dap_s3_not_exist_uses_correct_prefix_format() -> None:
    s3 = MagicMock()
    s3.list_objects_v2.return_value = {"KeyCount": 0}

    duplicate_check.check_dap_s3_not_exist(
        reporting_date=date(2026, 12, 1),
        s3=s3,
        bucket=DAP_BUCKET,
        logger=MagicMock(),
    )

    call_kwargs = s3.list_objects_v2.call_args.kwargs
    assert call_kwargs["Prefix"] == "reporting_date=2026-12-01/"

def test_check_downstream_not_exist_calls_os_and_dap_checks(monkeypatch) -> None:
    check_os = MagicMock()
    check_dap = MagicMock()
    monkeypatch.setattr(duplicate_check, "check_os_not_exist", check_os)
    monkeypatch.setattr(duplicate_check, "check_dap_s3_not_exist", check_dap)

    s3 = MagicMock()
    dynamodb = MagicMock()

    duplicate_check.check_downstream_not_exist(
        reporting_date=REPORTING_DATE,
        dynamodb=dynamodb,
        s3=s3,
        env=ENV,
        dap_bucket=DAP_BUCKET,
        logger=MagicMock(),
    )

    check_os.assert_called_once_with(
        reporting_date=REPORTING_DATE,
        dynamodb=dynamodb,
        env=ENV,
        logger=check_os.call_args.kwargs["logger"],
    )
    check_dap.assert_called_once_with(
        reporting_date=REPORTING_DATE,
        s3=s3,
        bucket=DAP_BUCKET,
        logger=check_dap.call_args.kwargs["logger"],
    )


def test_check_downstream_not_exist_propagates_os_error(monkeypatch) -> None:
    monkeypatch.setattr(
        duplicate_check,
        "check_os_not_exist",
        MagicMock(side_effect=AlreadyExistsError("OS data exists")),
    )
    monkeypatch.setattr(duplicate_check, "check_dap_s3_not_exist", MagicMock())

    with pytest.raises(AlreadyExistsError, match="OS data exists"):
        duplicate_check.check_downstream_not_exist(
            reporting_date=REPORTING_DATE,
            dynamodb=MagicMock(),
            s3=MagicMock(),
            env=ENV,
            dap_bucket=DAP_BUCKET,
            logger=MagicMock(),
        )
