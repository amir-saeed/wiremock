def test_main_loads_context_and_runs_selected_pipeline(monkeypatch) -> None:
    from jobs import maintenance_job

    job_context = object()
    pipeline = MagicMock()
    pipeline_cls = MagicMock(return_value=pipeline)
    pipeline_spec = SimpleNamespace(
        pipeline_cls=pipeline_cls,
        job_context_model=object,
    )
    load_job_context = MagicMock(return_value=job_context)

    monkeypatch.setattr(
        maintenance_job,
        "parse_args",
        MagicMock(
            return_value=SimpleNamespace(
                run_id="run-local-001",
                env="local",
                reporting_date="2026-06-08",
                aws_region="eu-west-1",
                es_host="localhost",
                es_port=5432,
                es_database="entity_store",
                es_user="motor_eqd",
                es_secret_name=None,
            )
        ),
    )
    monkeypatch.setattr(maintenance_job, "load_job_context", load_job_context)
    monkeypatch.setattr(
        maintenance_job,
        "PIPELINES",
        {(maintenance_job.JobName.MAINTENANCE, None): pipeline_spec},
    )

    maintenance_job.main()

    load_job_context.assert_called_once()
    runtime_args = load_job_context.call_args.kwargs["glue_runtime_args"]
    assert runtime_args["job_name"] == "maintenance"
    assert runtime_args["env"] == "local"
    assert runtime_args["run_id"] == "run-local-001"
    assert runtime_args["reporting_date"] == date(2026, 6, 8)
    assert isinstance(runtime_args["execution_ts"], datetime)
    assert runtime_args["execution_ts"].tzinfo is UTC
    assert runtime_args["aws_region"] == "eu-west-1"
    assert runtime_args["es"] == {
        "host": "localhost",
        "port": 5432,
        "database": "entity_store",
        "user": "motor_eqd",
        "secret_name": None,
    }
    pipeline_cls.assert_called_once_with()
    pipeline.run.assert_called_once_with(job_context)
