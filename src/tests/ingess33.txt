"""Tests for src/pipelines/maintenance/partitions.py."""
from datetime import date
from unittest.mock import MagicMock, call

import pytest

from pipelines.maintenance.partitions import ensure_partitions

# ---------------------------------------------------------------------------
# Shared fixtures
# ---------------------------------------------------------------------------

REPORTING_DATE = date(2026, 5, 12)


def _run(
    reporting_date: date = REPORTING_DATE,
    created: bool = False,
) -> tuple[MagicMock, MagicMock]:
    """Run ensure_partitions and return (postgres, logger) mocks."""
    postgres = MagicMock()
    postgres.ensure_quote_partition.return_value = created
    logger = MagicMock()

    ensure_partitions(
        reporting_date=reporting_date,
        postgres=postgres,
        logger=logger,
    )

    return postgres, logger


# ---------------------------------------------------------------------------
# Partition name format
# ---------------------------------------------------------------------------

def test_partition_name_standard_date() -> None:
    postgres, _ = _run(reporting_date=date(2026, 5, 12))
    partition_name = postgres.ensure_quote_partition.call_args.kwargs["partition_name"]
    assert partition_name == "quote_2026_05"


def test_partition_name_single_digit_month_is_zero_padded() -> None:
    postgres, _ = _run(reporting_date=date(2026, 1, 1))
    partition_name = postgres.ensure_quote_partition.call_args.kwargs["partition_name"]
    assert partition_name == "quote_2026_01"


def test_partition_name_december() -> None:
    postgres, _ = _run(reporting_date=date(2026, 12, 31))
    partition_name = postgres.ensure_quote_partition.call_args.kwargs["partition_name"]
    assert partition_name == "quote_2026_12"


def test_partition_name_uses_year_from_reporting_date() -> None:
    postgres, _ = _run(reporting_date=date(2025, 3, 1))
    partition_name = postgres.ensure_quote_partition.call_args.kwargs["partition_name"]
    assert partition_name.startswith("quote_2025_")


def test_partition_name_prefix_is_quote() -> None:
    postgres, _ = _run()
    partition_name = postgres.ensure_quote_partition.call_args.kwargs["partition_name"]
    assert partition_name.startswith("quote_")


# ---------------------------------------------------------------------------
# Postgres interaction
# ---------------------------------------------------------------------------

def test_ensure_quote_partition_called_once() -> None:
    postgres, _ = _run()
    postgres.ensure_quote_partition.assert_called_once()


def test_ensure_quote_partition_receives_reporting_date() -> None:
    postgres, _ = _run(reporting_date=REPORTING_DATE)
    assert postgres.ensure_quote_partition.call_args.kwargs["reporting_date"] == REPORTING_DATE


def test_ensure_quote_partition_receives_correct_partition_name() -> None:
    postgres, _ = _run(reporting_date=date(2026, 5, 12))
    assert postgres.ensure_quote_partition.call_args.kwargs["partition_name"] == "quote_2026_05"


# ---------------------------------------------------------------------------
# Branch: partition created
# ---------------------------------------------------------------------------

def test_logs_created_when_partition_is_new() -> None:
    _, logger = _run(created=True)
    log_messages = [c.args[0] for c in logger.info.call_args_list]
    assert any("created" in msg.lower() for msg in log_messages)


def test_logs_created_message_contains_partition_name() -> None:
    _, logger = _run(reporting_date=date(2026, 5, 12), created=True)
    log_messages = [c.args[0] for c in logger.info.call_args_list]
    assert any("quote_2026_05" in msg for msg in log_messages)


def test_does_not_log_already_exists_when_partition_is_new() -> None:
    _, logger = _run(created=True)
    log_messages = [c.args[0] for c in logger.info.call_args_list]
    assert not any("already exists" in msg.lower() for msg in log_messages)


# ---------------------------------------------------------------------------
# Branch: partition already exists
# ---------------------------------------------------------------------------

def test_logs_already_exists_when_partition_present() -> None:
    _, logger = _run(created=False)
    log_messages = [c.args[0] for c in logger.info.call_args_list]
    assert any("already exists" in msg.lower() for msg in log_messages)


def test_logs_already_exists_message_contains_partition_name() -> None:
    _, logger = _run(reporting_date=date(2026, 5, 12), created=False)
    log_messages = [c.args[0] for c in logger.info.call_args_list]
    assert any("quote_2026_05" in msg for msg in log_messages)


def test_does_not_log_created_when_partition_already_exists() -> None:
    _, logger = _run(created=False)
    # "created" should not appear as a standalone action — only in "already exists" context
    log_messages = [c.args[0] for c in logger.info.call_args_list]
    assert not any(
        msg.lower().strip().endswith("created") or "partition created" in msg.lower()
        for msg in log_messages
    )


# ---------------------------------------------------------------------------
# Logging — general
# ---------------------------------------------------------------------------

def test_logs_at_least_twice() -> None:
    _, logger = _run()
    assert logger.info.call_count >= 2


def test_initial_log_contains_partition_name() -> None:
    _, logger = _run(reporting_date=date(2026, 5, 12))
    first_log = logger.info.call_args_list[0].args[0]
    assert "quote_2026_05" in first_log
